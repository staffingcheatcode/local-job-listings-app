function renderTracker(){
  const j=JOBS.find(x=>x.id===state.trackJob);
  document.getElementById('trackTitle').textContent=j.title;
  document.getElementById('trackCo').textContent=`${j.co} · ${j.city}`;
  const pct=Math.round((state.stage/5)*100);
  document.getElementById('trackPct').textContent=pct+'%';

  let html='';
  for(let i=0;i<STAGES.length;i++){
    let cls='';
    if(i<state.stage) cls='done';
    else if(i===state.stage && state.stage<5) cls='current';
    else if(i===state.stage && state.stage>=5) cls='done';
    // final node handled by outcome card; show as done if reached
    const isLast = i===5;
    const label = isLast ? (state.stage>=5 ? (state.outcome==='offer'?'Offer extended':'Decision made') : 'Decision') : STAGES[i];
    let icon = '○';
    if(cls==='done') icon='<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3"><path d="M4 12l5 5L20 6"/></svg>';
    else if(cls==='current') icon='<b style="font-family:Barlow Condensed">'+(i+1)+'</b>';
    else icon='<span style="font-family:Barlow Condensed;font-weight:700">'+(i+1)+'</span>';
    const when = cls==='done' ? '<div class="when">'+fakeTime(i)+'</div>' : (cls==='current'?'<div class="when">In progress</div>':'');
    const nowBadge = cls==='current' ? '<span class="badge-now">Happening now</span>' : '';
    const lineHTML = isLast ? '' : `<div class="line"></div>`;
    html+=`<div class="step ${cls}">
      <div class="rail"><div class="node">${icon}</div>${lineHTML}</div>
      <div class="step-body">
        <h4>${label}</h4>
        ${ (cls==='done'||cls==='current') && STAGE_DESC[i] ? '<div class="when" style="color:#3c4d62;margin-top:2px">'+STAGE_DESC[i]+'</div>':''}
        ${when}${nowBadge}
      </div>
    </div>`;
  }
  document.getElementById('stepper').innerHTML=html;

  // outcome card
  const ow=document.getElementById('outcomeWrap');
  if(state.stage>=5){
    if(state.outcome==='offer'){
      ow.innerHTML=`<div class="outcome offer">
        <h3>🎉 Offer extended!</h3>
        <p>${j.co} would like to move forward. Check your messages to review details and next steps.</p>
        <button class="btn btn-primary mt-s" style="font-size:15px;padding:13px" onclick="go('messages')">View message</button>
      </div>`;
    } else {
      ow.innerHTML=`<div class="outcome reject">
        <h3>Not selected this time</h3>
        <p>The employer chose to move forward with other candidates. This happens often and isn't a reflection of your worth — keep applying.</p>
        <span class="reason-chip">Reason: Position filled</span>
        <button class="btn btn-primary mt-s" style="font-size:15px;padding:13px" onclick="go('feed')">Find similar jobs nearby</button>
      </div>`;
    }
    document.getElementById('trackEta').textContent = state.outcome==='offer'?'You have an offer — respond soon!':'Decision posted. Similar jobs are waiting.';
  } else {
    ow.innerHTML='';
    document.getElementById('trackEta').textContent='Most employers respond within 48 hours.';
  }
}
function fakeTime(i){
  const t=['Today, 9:14 AM','Today, 10:02 AM','Today, 1:30 PM','Yesterday','Yesterday'];
  return t[i]||'';
}
